Extending CallBack (java)



import com.phonegap.api.Plugin;
import com.phonegap.api.PluginResult;

public class WebIntent extends Plugin {
    @Override
    public PluginResult execute(String action, JSONArray args, String callbackId) {
            // Parse the arguments
            JSONObject obj = args.getJSONObject(0);
            String type = obj.has("type") ? obj.getString("type") : null;
            Uri uri = obj.has("url") ? Uri.parse(obj.getString("url")) : null;

            // open the intent, send result to JS
            openActivity(uri, type);
            return new PluginResult(PluginResult.Status.OK);
    }

    void openActivity(Uri uri, String type) {
        Intent i = (uri != null ? new Intent(action, uri) : new Intent(action));
        if (type != null) {
            i.setDataAndType(uri, type);
        }
        i.setFlags(Intent.FLAG_ACTIVITY_NEW_TASK);
        this.ctx.startActivity(i);
    }
}

Extending CallBack (javascript)



var WebIntent = function() {};

WebIntent.prototype.openActivity = function(params, success, fail) {
    return PhoneGap.exec(function(args) {
        success(args);
    }, function(args) {
        fail(args);
    }, 'WebIntent', 'openActivity', [params]);
};
